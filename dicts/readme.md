# python - dict

* key value map
  * Keys must be unique and hasable
* unordered
* mutable

# define dict

* dict()
* {key:value}
* dict.fromkeys([key], default)

# add  key 

* d[key] = value
* d.setdefault()
* d.update(key=value)

# remove key(s)

* del d[key]
* d.clear()
* d.pop(key)
* d.popitem()

# get key value

* d.get()
* d[key]

# copy dict

* d.copy()

# dict methods

* keys()
* values()
* items()
